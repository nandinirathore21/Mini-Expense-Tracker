// DOM Elements
const expenseNameInput = document.getElementById('expense-name');
const expenseAmountInput = document.getElementById('expense-amount');
const addBtn = document.getElementById('add-btn');
const expenseList = document.getElementById('expense-list');
const totalVal = document.getElementById('total-val');

// Keep a baseline reference to the empty message element
let emptyMsg = document.getElementById('empty-msg');

// State Array to hold items
let expenses = [];

// Function to update the DOM
function updateUI() {
    // Clear current list items
    expenseList.innerHTML = '';

    if (expenses.length === 0) {
        // If emptyMsg element was removed from the DOM, recreate it dynamically
        if (!emptyMsg) {
            emptyMsg = document.createElement('li');
            emptyMsg.className = 'empty-state';
            emptyMsg.id = 'empty-msg';
            emptyMsg.textContent = 'No expenses added yet.';
        }
        expenseList.appendChild(emptyMsg);
        totalVal.textContent = '0.00';
        return;
    }

    let total = 0;

    expenses.forEach((expense) => {
        total += expense.amount;

        // Create Item Container
        const li = document.createElement('li');
        li.className = 'expense-item';

        // Setup internal content structure
        li.innerHTML = `
            <div class="expense-info">
                <span class="expense-name">${expense.name}</span>
            </div>
            <div style="display: flex; align-items: center;">
                <span class="expense-cost">$${expense.amount.toFixed(2)}</span>
                <button class="btn-delete" data-id="${expense.id}">Delete</button>
            </div>
        `;

        expenseList.appendChild(li);
    });

    // Update Total Value
    totalVal.textContent = total.toFixed(2);

    // Attach event listeners to newly created delete buttons
    attachDeleteListeners();
}

// Function to handle adding an expense
function addExpense() {
    const name = expenseNameInput.value.trim();
    const amount = parseFloat(expenseAmountInput.value);

    // Simple Validation
    if (name === '' || isNaN(amount) || amount <= 0) {
        alert('Please enter a valid expense description and positive numeric amount.');
        return;
    }

    // Create object item structure
    const expense = {
        id: Date.now(), // Unique ID using timestamp
        name: name,
        amount: amount
    };

    expenses.push(expense);
    
    // Clear Inputs
    expenseNameInput.value = '';
    expenseAmountInput.value = '';

    updateUI();
}

// Helper function to manage delete event listeners safely
function attachDeleteListeners() {
    const deleteButtons = document.querySelectorAll('.btn-delete');
    deleteButtons.forEach(button => {
        button.addEventListener('click', function() {
            const idToIdenitfy = parseInt(this.getAttribute('data-id'));
            expenses = expenses.filter(expense => expense.id !== idToIdenitfy);
            updateUI();
        });
    });
}

// Event Listeners
addBtn.addEventListener('click', addExpense);

// Allow pressing "Enter" on input fields to submit
expenseAmountInput.addEventListener('keypress', function(e) {
    if (e.key === 'Enter') addExpense();
});
